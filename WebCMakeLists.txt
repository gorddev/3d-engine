#define OLAF_APPSTATE_OBJ
#include "core/factory/shapes/ShapeAPI.hpp"
#include "core/factory/shapes/shape-apis/Box3D.hpp"
#include "olaf-api/olaf_callbacks.hpp"

struct AppState {
    std::vector<gan::ShapeAPI> shapes;
};

AppState* g_main(gan::Book& book, int argc, char* argv[]) {

    auto& game = *new AppState();

    game.shapes.push_back(book.shapes.make<gan::Box3D>(3.0, 3.0, 3.0));

    return &game;
}

int g_iterate(gan::Book& book, AppState* game) {

    for (auto& s : game->shapes) {
        book.render.draw3D(book.camera, s);
        GL_CHECK();
    }
    return 0;
}
